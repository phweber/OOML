39 
40  Component cylinder = Cylinder(data.effectiveRadius*0.6,data.baseHeight+4 , 6 , false);
41  cylinder.rotate(0,90,0).translate(-data.baseHeight/2-2,0,data.baseThickness+data.effectiveRadius*1.2);
42 
43  Component cylinder2 = Cylinder(data.effectiveRadius*0.85,data.effectiveRadius/2,75,false);
44 
45  cylinder2.translate(0,0,data.baseThickness);
46 
47  Component cutout = sphere + cube + cylinder + cylinder2;
48 
49 
50  // Support
51  double support_height = data.totalHeight - (data.baseThickness+data.effectiveRadius*1.4);
52  if (support_height < 0){
53  Component spherewheelsupport = main_shape - cutout;
54  return spherewheelsupport;
55  }
56 
57 //else
58  Component support = Cylinder(data.baseHeight/2,support_height+1,75,false);
59  Component spherewheelsupport = support
60  + (main_shape - cutout).translate(0,0,support_height)
61  - Cylinder(data.effectiveRadius*0.80,data.totalHeight,75,false).translate(0,0,6);
62 
63  return spherewheelsupport;
