23 TransformMatrix TransformMatrix::getInv() const {
24  TransformMatrix inv;
25 
26  //Traspose the 3x3 submatrix
27  for (int i=1;i<=3;i++){
28  for (int j=1;j<=3;j++){
29  inv.set(i,j,get(j,i));
30  }
31  }
32 
33  //Las row all 0 but last
34  for (int i=1;i<=3;i++)
35  inv.set(4,i,0);
36  inv.set(4,4,1);
37 
38  double e14 = get(1,4)*get(1,1) + get(2,4)*get(2,1) + get(3,4)*get(3,1) ;
39  double e24 = get(1,4)*get(1,2) + get(2,4)*get(2,2) + get(3,4)*get(3,2) ;
40  double e34 = get(1,4)*get(1,3) + get(2,4)*get(2,3) + get(3,4)*get(3,3) ;
41 
42  inv.set(1,4,-e14);
43  inv.set(2,4,-e24);
44  inv.set(3,4,-e34);
45 
46  return inv;
47
