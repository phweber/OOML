27 Component & Component::moveToLink(const Component & base, int link_id){

29  RefSys link = base.getLink(link_id);
30  TransformMatrix trans = link.getTransformMatrix();

32  double xa, ya, za; trans.getGlobalXYZAngles(xa,ya,za);
